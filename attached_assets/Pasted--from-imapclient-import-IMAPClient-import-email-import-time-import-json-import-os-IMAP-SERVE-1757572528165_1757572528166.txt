
from imapclient import IMAPClient
import email
import time
import json
import os

IMAP_SERVER = "imap.gmail.com"
EMAIL_USER = "shahanedhananjay0@gmail.com"
EMAIL_PASSWORD = "zclp zwex hqkr fvqp"

OUTPUT_JSON = "emails.json"
ATTACHMENTS_DIR = "email-attachments"   # ✅ changed folder name

# Create attachments folder if not exists
os.makedirs(ATTACHMENTS_DIR, exist_ok=True)

# Initialize JSON file if not exists
if not os.path.exists(OUTPUT_JSON):
    with open(OUTPUT_JSON, "w", encoding="utf-8") as f:
        json.dump([], f, indent=4)


def save_email_json(entry):
    """Append email entry to JSON file if UID not already saved"""
    with open(OUTPUT_JSON, "r+", encoding="utf-8") as f:
        data = json.load(f)
        if not any(d["uid"] == entry["uid"] for d in data):  # ✅ prevent duplicates
            data.append(entry)
            f.seek(0)
            json.dump(data, f, indent=4, ensure_ascii=False)


def process_new_email(uid, msg):
    """Extract details and save to JSON"""
    subject = msg.get("Subject", "No Subject")
    from_ = msg.get("From", "Unknown")

    # --- Extract plain text body ---
    body = ""
    if msg.is_multipart():
        for part in msg.walk():
            if part.get_content_type() == "text/plain" and not part.get_filename():
                try:
                    body = part.get_payload(decode=True).decode("utf-8", errors="ignore")
                except Exception:
                    pass
    else:
        try:
            body = msg.get_payload(decode=True).decode("utf-8", errors="ignore")
        except Exception:
            pass

    # --- Handle attachments ---
    attachments = []
    if msg.is_multipart():
        for part in msg.walk():
            filename = part.get_filename()
            if filename:
                filepath = os.path.join(ATTACHMENTS_DIR, filename)
                with open(filepath, "wb") as f:
                    f.write(part.get_payload(decode=True))
                attachments.append(filename)

    entry = {
        "uid": uid,
        "from": from_,
        "subject": subject,
        "body": body.strip(),
        "attachments": attachments,
        "replystatus": "pending"
    }

    print(f"📧 Delivered Email UID {uid} | From: {from_} | Subject: {subject}")
    save_email_json(entry)


def main():
    with IMAPClient(IMAP_SERVER) as client:
        client.login(EMAIL_USER, EMAIL_PASSWORD)
        client.select_folder("INBOX")

        print("📡 Mailbox watcher started...")

        # ✅ Baseline = newest UID at startup (ignore old emails)
        all_messages = client.search("ALL")
        last_uid = max(all_messages) if all_messages else 0
        print(f"👉 Ignoring old emails, starting from UID {last_uid}")

        while True:
            client.noop()  # keep connection alive

            # Get all UIDs and only process new ones
            all_messages = client.search("ALL")
            new_messages = [uid for uid in all_messages if uid > last_uid]

            if new_messages:
                for uid in sorted(new_messages):
                    msg_data = client.fetch([uid], ["RFC822"])
                    msg = email.message_from_bytes(msg_data[uid][b"RFC822"])
                    process_new_email(uid, msg)
                    last_uid = max(last_uid, uid)  # ✅ update baseline
            else:
                print("✅ No new delivered emails found.")

            print("⏳ Waiting 30 seconds before next check...")
            time.sleep(30)


if __name__ == "__main__":
    main()
