Here is a complete, detailed, and robust Python script for a **real-time email agent** that:

- Checks new emails every 20 seconds via **IMAP**
- Extracts email data (sender, subject, body, attachments)
- Uses **LangChain** with an LLM to generate reply content
- Sends replies using **SMTP**
- Saves each processed email and generated reply as JSON files

This serves as a practical, near real-time automation pipeline combining LangChain + IMAP + SMTP + JSON storage.

***

```python
import imaplib
import email
from email.header import decode_header
import json
import smtplib
from email.mime.text import MIMEText
import time
import os
from langchain.llms import OpenAI
from langchain.chains import LLMChain
from langchain.prompts import PromptTemplate
from dotenv import load_dotenv

# Load configuration from .env file
load_dotenv()

IMAP_HOST = os.getenv("EMAIL_IMAP_HOST", "imap.gmail.com")
SMTP_HOST = os.getenv("EMAIL_SMTP_HOST", "smtp.gmail.com")
EMAIL_ADDRESS = os.getenv("EMAIL_ADDRESS")
EMAIL_PASSWORD = os.getenv("EMAIL_APP_PASSWORD")
OPENAI_API_KEY = os.getenv("OPENAI_API_KEY")

class EmailFetcher:
    def __init__(self, host, user, password):
        self.host = host
        self.user = user
        self.password = password
        self.mail = None

    def connect(self):
        if self.mail is not None:
            try:
                self.mail.logout()
            except:
                pass
        self.mail = imaplib.IMAP4_SSL(self.host)
        self.mail.login(self.user, self.password)
        self.mail.select("inbox")

    def fetch_unseen_emails(self):
        emails = []
        try:
            status, messages = self.mail.search(None, 'UNSEEN')
            if status != "OK":
                return emails
            email_ids = messages[0].split()
            for e_id in email_ids:
                status, msg_data = self.mail.fetch(e_id, '(RFC822)')
                if status != "OK":
                    continue
                for response_part in msg_data:
                    if isinstance(response_part, tuple):
                        msg = email.message_from_bytes(response_part[1])
                        sender = msg['From']
                        subject = decode_header(msg['Subject'])[0][0]
                        if isinstance(subject, bytes):
                            subject = subject.decode(errors='ignore')
                        body = ""
                        attachments = []
                        if msg.is_multipart():
                            for part in msg.walk():
                                content_disposition = part.get("Content-Disposition", "")
                                if part.get_content_type() == "text/plain" and "attachment" not in content_disposition:
                                    try:
                                        body += part.get_payload(decode=True).decode()
                                    except:
                                        pass
                                elif "attachment" in content_disposition:
                                    filename = part.get_filename()
                                    if filename:
                                        attachments.append(filename)
                        else:
                            try:
                                body = msg.get_payload(decode=True).decode()
                            except:
                                pass
                        email_data = {
                            "sender": sender,
                            "subject": subject,
                            "body": body,
                            "attachments": attachments
                        }
                        emails.append(email_data)
        except Exception as e:
            print(f"Error fetching unseen emails: {e}")
        return emails

class EmailSender:
    def __init__(self, smtp_host, user, password):
        self.smtp_host = smtp_host
        self.user = user
        self.password = password

    def send_email(self, to_address, subject, body):
        msg = MIMEText(body)
        msg['From'] = self.user
        msg['To'] = to_address
        msg['Subject'] = subject

        try:
            with smtplib.SMTP(self.smtp_host, 587) as server:
                server.starttls()
                server.login(self.user, self.password)
                server.sendmail(self.user, to_address, msg.as_string())
            print(f"Sent email to {to_address} with subject '{subject}'")
        except Exception as e:
            print(f"Error sending email to {to_address}: {e}")

def save_email_as_json(email_dict, directory="emails"):
    os.makedirs(directory, exist_ok=True)
    safe_subject = email_dict['subject'][:50].replace(" ", "_").replace("/", "_").replace("\\", "_")
    filename = f"{directory}/{safe_subject}.json"
    try:
        with open(filename, "w", encoding="utf-8") as f:
            json.dump(email_dict, f, indent=4, ensure_ascii=False)
        print(f"Saved email to {filename}")
    except Exception as e:
        print(f"Error saving json for email '{email_dict['subject']}': {e}")

# Set up LangChain LLM with OpenAI
template = """You received an email from: {sender}
Subject: {subject}
Body: {body}

Generate a professional, concise response."""
prompt = PromptTemplate(input_variables=["sender", "subject", "body"], template=template)
llm = OpenAI(openai_api_key=OPENAI_API_KEY, temperature=0.3)
chain = LLMChain(llm=llm, prompt=prompt)

def main():
    fetcher = EmailFetcher(IMAP_HOST, EMAIL_ADDRESS, EMAIL_PASSWORD)
    sender = EmailSender(SMTP_HOST, EMAIL_ADDRESS, EMAIL_PASSWORD)

    while True:
        try:
            fetcher.connect()
            new_emails = fetcher.fetch_unseen_emails()
            if new_emails:
                for email_data in new_emails:
                    print(f"Processing email from {email_data['sender']} with subject '{email_data['subject']}'")
                    reply = chain.run(sender=email_data["sender"], subject=email_data["subject"], body=email_data["body"])
                    email_data["reply"] = reply

                    # Send reply email
                    sender.send_email(email_data["sender"], "Re: " + email_data["subject"], reply)

                    # Save original email data + reply to JSON
                    save_email_as_json(email_data)

            fetcher.mail.logout()
        except Exception as e:
            print(f"Error in main loop: {e}")

        # Poll every 20 seconds for near real-time checking
        time.sleep(20)

if __name__ == "__main__":
    main()
```

***

## Explanation:

- **Real-time polling every 20 seconds** is ensured by the `while True` loop combined with `time.sleep(20)`.
- The **IMAP connection is recreated every cycle** using `.connect()` and logged out after fetching for stability.
- **Unseen emails** (new emails) are retrieved and parsed for sender, subject, body, and attachments.
- **LangChain LLM** generates a professional reply based on email content.
- The reply is sent using **SMTP**.
- The full email data, including the generated reply, is saved as a **JSON** file in the "emails" directory.
- Program includes **error handling** to prevent crashes and print helpful messages.
